# Bookworm/Trixie (KMS) panel config for MZP351HV00TR (480x320 DPI).
# Manufacturer's exact overlay/order. Pairs with config_append.txt and
# BIGA_SDL_VIDEO=kmsdrm. setup.py copies this verbatim to /boot/firmware/.
#
# Order matters: spi0-0cs frees the CS pin BEFORE ads7846 touch loads, then
# KMS loads with no GPIO conflict.

dtoverlay=spi0-0cs
dtoverlay=ads7846,penirq=27,swapxy=1,xmin=180,xmax=3900,ymin=180,ymax=3900
dtoverlay=vc4-kms-v3d
max_framebuffers=2
dtoverlay=vc4-kms-dpi-generic
dtparam=hactive=480,hfp=20,hsync=10,hbp=10
dtparam=vactive=320,vfp=10,vsync=2,vbp=2
dtparam=clock-frequency=12000000
dtparam=hsync-invert,vsync-invert,pixclk-invert
dtparam=rgb565-padhi
dtparam=backlight-gpio=18
gpio=18=op,dh,pd
